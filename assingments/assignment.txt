import java.util.*;


//MutualFund is a dat atype
class MutualFund {
    private String fundName;
    private String fundType;
    private double nav; // Net Asset Value

    public MutualFund(String fundName, String fundType, double nav) {
        this.fundName = fundName;
        this.fundType = fundType;
        this.nav = nav;
    }


 
    public String getFundName() {
        return fundName;
    }

    public String getFundType() {
        return fundType;
    }

    public double getNav() {
        return nav;
    }

    public void setNav(double nav) {
        this.nav = nav;
    }

    
    public String toString() {
        return "MutualFund{" +
                "fundName='" + fundName + '\'' +
                ", fundType='" + fundType + '\'' +
                ", nav=" + nav +
                '}';
    }
}




class Investment {
    private MutualFund mutualFund;
    private int units;

    public Investment(MutualFund mutualFund, int units) {
        this.mutualFund = mutualFund;
        this.units = units;
    }

    public MutualFund getMutualFund() {
        return mutualFund;
    }

    public int getUnits() {
        return units;
    }

    public double getValue() {
        return units * mutualFund.getNav();
    }

    @Override
    public String toString() {
        return "Investment{" +
                "mutualFund=" + mutualFund +
                ", units=" + units +
                ", value=" + getValue() +
                '}';
    }
}

class Portfolio {
    private String investorName;
    private List<Investment> investments;

    public Portfolio(String investorName) {
        this.investorName = investorName;
        this.investments = new ArrayList<>();
    }

    public void addInvestment(MutualFund mutualFund, int units) {
        investments.add(new Investment(mutualFund, units));
    }

    public double getTotalValue() {
        double totalValue = 0;
        for (Investment investment : investments) {
            totalValue += investment.getValue();
        }
        return totalValue;
    }

    @Override
    public String toString() {
        return "Portfolio{" +
                "investorName='" + investorName + '\'' +
                ", investments=" + investments +
                ", totalValue=" + getTotalValue() +
                '}';
    }
}
